# Quickstart

:::info Before Getting Started

```jsx
import { DyteMeeting } from '@dytesdk/mobile';

function App() {
  return (
    <DyteMeeting
      onInit={(meeting) => {}}
      clientId={`orgId || clientId`}
      meetingConfig={{
        roomName: `roomName`,
        authToken: `authToken`,
      }}
    />
  );
}

export default App;
```
